这个程序使用random模块从列表中随机选取一个数字。

废话不多说直接上代码

#-*-coding:utf8;-*-
#在python3控制台里运行
#导入随机选取模块
import random
#生成并存储第一位数字
a=random.choice(['1','2','3','4','5','6','7','8','9','0'])
#生成并存储第二位数字
b=random.choice(['1','2','3','4','5','6','7','8','9','0'])
#生成并存储第三位数字
c=random.choice(['1','2','3','4','5','6','7','8','9','0'])
#生成并存储第四位数字
d=random.choice(['1','2','3','4','5','6','7','8','9','0'])
#打印出已经存储的验证码
print("验证码为：",a,b,c,d)
